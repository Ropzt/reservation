select t.route_id, t.service_id, t.trip_id, t.trip_headsign, 
r.route_long_name, 
st.arrival_time, st.departure_time, st.stop_id, st.stop_sequence, 
s.stop_name, s.parent_station,
c.tuesday, c.start_date, c.end_date
from trips as t,
routes as r, 
stop_times as st,
stops as s,
calendar as c
where t.service_id = c.service_id
and st.stop_id = s.stop_id
and t.trip_id=st.trip_id 
and t.route_id=r.route_id 
and t.trip_id == "OCESN008860F2802887246"
 ;
